Reverse Num :

import java.util.Scanner;

class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int[] a = new int[1]; // acts like pointer
        a[0] = sc.nextInt();

        int num = a[0];
        int rev = 0;

        while (num != 0) {
            int digit = num % 10;
            rev = rev * 10 + digit;
            num = num / 10;
        }

        System.out.println("The new number = " + rev);
    }
}
